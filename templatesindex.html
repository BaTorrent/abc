<script>
    async function updateDashboard() {
        try {
            const response = await fetch('/api/data');
            const data = await response.json();

            if (data.status === "success") {
                // ১. ওপরের ড্যাশবোর্ড বক্স আপডেট
                document.getElementById('buy-exchange').innerText = data.best_buy.toUpperCase();
                document.getElementById('buy-price').innerText = `$${data.buy_price.toFixed(4)}`;
                document.getElementById('sell-exchange').innerText = data.best_sell.toUpperCase();
                document.getElementById('sell-price').innerText = `$${data.sell_price.toFixed(4)}`;

                const resultContainer = document.getElementById('result-container');
                const resultText = document.getElementById('result-text');

                if (data.net_profit_usd > 0) {
                    resultContainer.className = "result-box opportunity-found";
                    resultText.innerHTML = `🚀 Opportunity Found! | Net Profit: $${data.net_profit_usd.toFixed(2)} (${data.net_profit_pct.toFixed(2)}%)`;
                } else {
                    resultContainer.className = "result-box no-opportunity";
                    resultText.innerText = "❌ No profitable gaps found after fee deductions.";
                }

                // ২. অল এক্সচেঞ্জ লাইভ টেবিল আপডেট লজিক
                const tableBody = document.getElementById('exchange-table-body');
                tableBody.innerHTML = ""; // আগের ডাটা ক্লিয়ার করা

                // অবজেক্টটিকে সর্ট করে বা সরাসরি টেবিলে লুপ চালানো
                for (const [name, info] of Object.entries(data.all_prices)) {
                    const row = `
                        <tr>
                            <td style="font-weight: 600; color: #fff;">${name.toUpperCase()}</td>
                            <td class="badge-buy">$${info.buy_at.toFixed(4)}</td>
                            <td class="badge-sell">$${info.sell_at.toFixed(4)}</td>
                        </tr>
                    `;
                    tableBody.innerHTML += row;
                }

                document.getElementById('last-updated').innerText = `Last checked: ${new Date().toLocaleTimeString()}`;
            }
        } catch (error) {
            console.error("Dashboard table error:", error);
        }
    }

    updateDashboard();
    setInterval(updateDashboard, 10000); // ১০ সেকেন্ড পর পর অটো আপডেট
</script>
